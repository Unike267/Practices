%
% A fairly complicated example from section 2.9 of the package
% documentation. This reproduces an example from Wikipedia:
% http://en.wikipedia.org/wiki/Gantt_chart
%
\definecolor{barblue}{RGB}{153,204,254}
\definecolor{groupblue}{RGB}{51,102,254}
\definecolor{linkred}{RGB}{165,0,33}
\renewcommand\sfdefault{phv}
\renewcommand\mddefault{mc}
\renewcommand\bfdefault{bc}
\setganttlinklabel{s-s}{START-TO-START}
\setganttlinklabel{f-s}{FINISH-TO-START}
\setganttlinklabel{f-f}{FINISH-TO-FINISH}
\sffamily
\begin{ganttchart}[
    canvas/.append style={fill=none, draw=black!5, line width=.75pt},
    hgrid style/.style={draw=black!15, line width=.75pt},
    vgrid={*1{draw=black!30, line width=.75pt}},
    today=35,
    today rule/.style={
      draw=none
    },
    today label = {},
    title/.style={draw=none, fill=none},
    title label font=\bfseries\footnotesize,
    title label node/.append style={below=7pt},
    include title in canvas=false,
    bar label font=\mdseries\small\color{blue!63},
    bar label node/.append style={left=2cm},
    bar/.append style={draw=none, fill=blue!63},
    bar incomplete/.append style={fill=barblue},
    bar progress label font=\mdseries\footnotesize\color{red!30},
    group incomplete/.append style={fill=groupblue},
    group left shift=0,
    group right shift=0,
    group height=.5,
    group peaks tip position=0,
    group label node/.append style={left=.6cm},
    group progress label font=\bfseries\small\color{red!0},
    link/.style={-latex, line width=1.5pt, linkred},
    link label font=\scriptsize\bfseries,
    link label node/.append style={below left=-2pt and 0pt}
  ]{1}{35}
  \gantttitle[
    title label node/.append style={below left=7pt and -3pt}
  ]{Semanas:\quad1}{1}
  \gantttitlelist{2,...,35}{1} \\
  \ganttgroup[progress label text={}]{Fase 1. Recursos SW/HW}{1}{35} \\
  \ganttbar[progress label text={},name=tarea1_1]{\textbf{Tarea 1}}{1}{35} \\
  \ganttbar[progress label text={},name=tarea2_1]{\textbf{Tarea 2}}{1}{35} \\
  [grid]
  \ganttgroup[progress label text={}]{Fase 2. Caracterización}{1}{20} \\
  \ganttbar[progress label text={},name=tarea1_2]{\textbf{Tarea 1}}{1}{4} \\
  \ganttbar[progress label text={},name=tarea2_2]{\textbf{Tarea 2}}{2}{7} \\
  \ganttbar[progress label text={},name=tarea3_2]{\textbf{Tarea 3}}{3}{11} \\
  \ganttbar[progress label text={},name=tarea4_2]{\textbf{Tarea 4}}{7}{13} \\
  \ganttbar[progress label text={},name=tarea5_2]{\textbf{Tarea 5}}{8}{12} \\
  \ganttbar[progress label text={},name=tarea6_2]{\textbf{Tarea 6}}{6}{15} \\
  \ganttbar[progress label text={},name=tarea7_2]{\textbf{Tarea 7}}{15}{20} \\
  [grid]
  \ganttgroup[progress label text={}]{Fase 3. Coprocesador IA}{24}{34} \\
  \ganttbar[progress label text={},name=tarea1_3]{\textbf{Tarea 1}}{24}{27} \\
  \ganttbar[progress label text={},name=tarea2_3]{\textbf{Tarea 2}}{32}{33} \\
  \ganttbar[progress label text={},name=tarea2_3]{\textbf{Tarea 3}}{33}{34} \\
  [grid]
  \ganttgroup[progress label text={}]{Fase 4. Documentación}{1}{35} \\
  \ganttbar[progress label text={},name=tarea1_4]{\textbf{Tarea 1}}{1}{35} \\
  \ganttbar[progress label text={},name=tarea2_4]{\textbf{Tarea 2}}{16}{33} \\
  \ganttbar[progress label text={},name=tarea2_4]{\textbf{Tarea 3}}{32}{35} \\
